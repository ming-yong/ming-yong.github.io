//grouping together
dogs.forEach(dog => {
 console.group(`${dog.name}`); //or groupCollapsed
 console.log(`This is ${dog.name}`);
 console.log(`${dog.name} is ${dog.age * 7} years old`);
 console.groupEnd(`${dog.name}`);
});

//timing

 });
